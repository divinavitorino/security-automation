pipeline {
    agent any
    options {
        ansiColor('gnome-terminal') 
    }
    stages {
        stage('Repository Clone') {
            steps{
              git branch: 'main', changelog: false, url: '[URL github repository]'
            }
        }
        stage ('Block user') {
            steps{
              ansiblePlaybook colorized: true, installation: 'ansible', inventory: '[inventory file]', limit: 'all', playbook: '[archivo de playbook]' 
            }
        }
        stage('Block Sleep') {
            steps{
                sleep time: "${time_sleep}", unit: 'DAYS'
            }
        }
        stage ('Unblock user') {
            steps{
              ansiblePlaybook colorized: true, installation: 'ansible', inventory: '[inventory file]', limit: 'all', playbook: '[playbook file]' 
            }
        }
    }
}
